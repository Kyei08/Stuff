Guessing Game

Guess a number between 1 and 100.

const titleElement = document.getElementById('title'); const messageElement = document.getElementById('message'); const guessInputElement = document.getElementById('guess-input'); const guessButton = document.getElementById('guess-button'); let randomNumber = Math.floor(Math.random() * 100) + 1; let attempts = 0; function checkGuess() { const guess = parseInt(guessInputElement.value); if (guess < randomNumber) { messageElement.textContent = 'Too low! Try again.'; } else if (guess > randomNumber) { messageElement.textContent = 'Too high! Try again.'; } else { attempts++; messageElement.textContent = `Congratulations! You guessed the number in ${attempts} attempts.`; titleElement.textContent = 'You Win!'; guessInputElement.disabled = true; guessButton.disabled = true; } } guessButton.addEventListener('click', checkGuess);
